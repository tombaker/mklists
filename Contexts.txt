class ExecutionContext:
    """Fully resolved execution state for a mklists run."""

    datadir_contexts: list[DatadirContext]
    passes: list[ExecutionPass]

    # repo-level files needed for backup directory initialization
    repo_configfile: Path | None
    repo_rulefile: Path | None

    # backup policy (with paths resolved)
    backup_rootdir: Path | None
    backup_depth: int

    # features (with paths resolved)
    routing_dict: dict[str, Path]
    htmldir: Path | None

ConfigContext - was: MklistsConfig
- config.py (still)

class ConfigContext:
    """Normalized, validated settings for processing one or more datadirs.

    A single config instance is derived from built-in defaults plus an optional
    config file. Multiple datadirs may share the same config instance when they
    share the same effective config file.
    """

    # Provenance
    configfile_used: Path | None
    config_rootdir: Path

    # Settings
    verbose: bool
    backup: BackupConfig
    routing: RoutingConfig
    safety: SafetyConfig
    urlify: UrlifyConfig


class BackupConfig:
    backup_enabled: bool
    backup_rootdir: Path
    backup_depth: int


class RoutingConfig:
    routing_enabled: bool
    routing_dict: dict[str, Path]


class UrlifyConfig:
    urlify_enabled: bool
    urlify_dir: Path
