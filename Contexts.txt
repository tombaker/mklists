---------------------------------------------------------
Structural Context - derived from filesystem structure
- StructuralContext          - resolve_structural_context $MKLSTRUCTURE/contexts_run.py
  - DatadirStructuralContext - $MKLSTRUCTURE/contexts_datadir.py

class DatadirContext:
    datadir: Path
    configfile_used: Path | None
    config_context: ConfigContext
    rules: list  # list[Rule]

class StructuralContext:
    config_rootdir: Path
    repo_configfile: Path | None
    repo_rulefile: Path | None
    datadir_contexts: list[DatadirContext]

---------------------------------------------------------
Execution Context - everything needed for execution
- ExecutionPlan           - resolve_execution_context $MKLMKL/execution/execution_context.py
  - DatadirPlan
  - PassPlan (per-pass execution unit)

class ExecutionPass:
    """One concrete execution pass."""

    backup_snapshot_dir: Path | None

class ExecutionPlan:
    """Fully resolved execution state for a mklists run."""

    datadir_contexts: list[DatadirContext]
    passes: list[ExecutionPass]

    # repo-level files needed for backup directory initialization
    repo_configfile: Path | None
    repo_rulefile: Path | None

    # backup policy (with paths resolved)
    backup_rootdir: Path | None
    backup_depth: int

    # features (with paths resolved)
    routing_dict: dict[str, Path]
    htmldir: Path | None

---------------------------------------------------------
ConfigContext

class ConfigContext:
    configfile_used: Path | None
    config_rootdir: Path
    verbose: bool
    backup: BackupConfig
    routing: RoutingConfig
    safety: SafetyConfig
    urlify: UrlifyConfig

class BackupConfig:
    backup_enabled: bool
    backup_rootdir: Path
    backup_depth: int

class RoutingConfig:
    routing_enabled: bool
    routing_dict: dict[str, Path]

class UrlifyConfig:
    urlify_enabled: bool
    urlify_dir: Path
